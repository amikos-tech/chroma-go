```go
func TestBuildContentEFFromConfig_ConsistentHashRoundTrip(t *testing.T) {
    // Step 1: create EF
    originalEF := embeddings.NewConsistentHashEmbeddingFunction()

    // Step 2: store via SetContentEmbeddingFunction (uses the capabilityAware mock pattern)
    // Since consistent_hash implements EmbeddingFunction, SetContentEmbeddingFunction should work
    // We need to use a ContentEmbeddingFunction that also implements EmbeddingFunction
    // Use the adapter to wrap the dense EF as content
    contentEF := embeddings.AdaptEmbeddingFunctionToContent(
        originalEF,
        embeddings.CapabilityMetadata{Modalities: []embeddings.Modality{embeddings.ModalityText}, SupportsBatch: true},
    )
    // But the adapter does NOT implement EmbeddingFunction, so SetContentEmbeddingFunction skips.
    // Instead, test the config round-trip via SetEmbeddingFunction directly:
    config := NewCollectionConfiguration()
    config.SetEmbeddingFunction(originalEF)

    // Step 3: build content EF from config
    rebuiltContentEF, err := BuildContentEFFromConfig(config)
    require.NoError(t, err)
    require.NotNil(t, rebuiltContentEF)

    // Step 4: verify capabilities
    ca, ok := rebuiltContentEF.(embeddings.CapabilityAware)
    require.True(t, ok)
    assert.True(t, ca.Capabilities().SupportsModality(embeddings.ModalityText))
}
```
