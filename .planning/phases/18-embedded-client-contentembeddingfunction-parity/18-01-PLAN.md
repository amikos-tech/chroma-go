**2. `Close()` method (line ~1405-1423):**
Replace the entire Close() method with sharing detection matching HTTP path (per D-03):

```go
func (c *embeddedCollection) Close() error {
    if !c.ownsEF.Load() {
        return nil
    }
    c.mu.RLock()
    ef := c.embeddingFunction
    contentEF := c.contentEmbeddingFunction
    c.mu.RUnlock()
    c.closeOnce.Do(func() {
        var errs []error
        if contentEF != nil {
            if closer, ok := contentEF.(io.Closer); ok {
                if err := safeCloseEF(closer); err != nil {
                    errs = append(errs, err)
                }
            }
        }
        if ef != nil {
            shared := false
            denseEF := unwrapCloseOnceEF(ef)
            if unwrapper, ok := contentEF.(embeddingspkg.EmbeddingFunctionUnwrapper); ok {
                shared = unwrapper.UnwrapEmbeddingFunction() == denseEF
            } else if efFromContent, ok := contentEF.(embeddingspkg.EmbeddingFunction); ok {
                shared = efFromContent == denseEF
            }
            if !shared {
                if closer, ok := ef.(io.Closer); ok {
                    if err := safeCloseEF(closer); err != nil {
                        errs = append(errs, err)
                    }
                }
            }
        }
        c.closeErr = stderrors.Join(errs...)
    })
    return c.closeErr
}
```

This replaces the old Close() that only handled denseEF with a full dual-EF sharing detection mirroring `CollectionImpl.Close()` in `collection_http.go:709-746`. Key differences from old code:
- Snapshots BOTH `ef` and `contentEF` under RLock (Pitfall 3 avoidance)
- Closes contentEF first via `safeCloseEF` (not bare closer.Close())
- Uses `unwrapCloseOnceEF` to unwrap before identity comparison
- Uses `stderrors.Join` for multiple errors instead of single error assignment
- Removes the old `defer recover` block since `safeCloseEF` handles panic recovery internally
